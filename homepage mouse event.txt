    <p>Hover over this section to change the background color!</p>
  </div>

  <div class="main-content">
    <p>Focus on the input below to change the background color:</p>
    <input type="text" placeholder="Click here..." onfocus="changeBackgroundOnFocus()" onblur="resetBackground()">
  </div>

  <script>
    // Change background color on mouse over
    function changeBackgroundOnHover() {
      document.body.style.backgroundColor = "#e0f7fa"; // Light blue
    }

    // Change background color on focus
    function changeBackgroundOnFocus() {
      document.body.style.backgroundColor = "#fff9c4"; // Light yellow
    }

    // Reset background color to default
    function resetBackground() {
      document.body.style.backgroundColor = ""; // Default color
    }
  </script>
