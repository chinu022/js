  <h2>Array Operations</h2>

  <button onclick="initializeArray()">Initialize Array</button>
  <button onclick="removeElement()">Remove Specific Element</button>
  <button onclick="checkElement()">Check if Array Contains Value</button>
  <button onclick="emptyArray()">Empty Array</button>
  <br><br>

  <p id="arrayDisplay"></p>
  <p id="resultDisplay"></p>

  <script>
    // Initialize an array with some values
    let myArray = [];

    function initializeArray() {
      myArray = [1, 2, 3, 4, 5, 6];
      displayArray();
    }

    // Function to display the array
    function displayArray() {
      document.getElementById("arrayDisplay").innerText = "Array: [" + myArray.join(", ") + "]";
      document.getElementById("resultDisplay").innerText = "";
    }

    // 1. Remove a specific element from the array
    function removeElement() {
      const element = prompt("Enter the element to remove:");
      const index = myArray.indexOf(Number(element));
      if (index !== -1) {
        myArray.splice(index, 1);
        document.getElementById("resultDisplay").innerText = `Element ${element} removed.`;
      } else {
        document.getElementById("resultDisplay").innerText = `Element ${element} not found in array.`;
      }
      displayArray();
    }

    // 2. Check if an array contains a specific value
    function checkElement() {
      const element = prompt("Enter the element to check:");
      if (myArray.includes(Number(element))) {
        document.getElementById("resultDisplay").innerText = `Array contains the element ${element}.`;
      } else {
        document.getElementById("resultDisplay").innerText = `Array does not contain the element ${element}.`;
      }
    }

    // 3. Empty the array
    function emptyArray() {
      myArray.length = 0;  // Set array length to 0 to empty it
      document.getElementById("resultDisplay").innerText = "Array has been emptied.";
      displayArray();
    }
  </script>
