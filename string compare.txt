  <script>
    function compareStrings() {
      const string1 = document.getElementById("string1").value;
      const string2 = document.getElementById("string2").value;
      let result = "";

      // Method 1: Check for exact equality
      if (string1 === string2) {
        result += "The strings are exactly equal.<br>";
      } else {
        result += "The strings are not exactly equal.<br>";
      }

      // Method 2: Check for equality ignoring case
      if (string1.toLowerCase() === string2.toLowerCase()) {
        result += "The strings are equal (case insensitive).<br>";
      } else {
        result += "The strings are not equal (case insensitive).<br>";
      }

      // Method 3: Check if one string includes the other
      if (string1.includes(string2)) {
        result += `The first string includes the second string.<br>`;
      } else if (string2.includes(string1)) {
        result += `The second string includes the first string.<br>`;
      } else {
        result += "Neither string includes the other.<br>";
      }

      // Method 4: Compare the lengths of the strings
      if (string1.length === string2.length) {
        result += "Both strings have the same length.<br>";
      } else if (string1.length > string2.length) {
        result += "The first string is longer than the second string.<br>";
      } else {
        result += "The second string is longer than the first string.<br>";
      }

      // Display the results
      document.getElementById("result").innerHTML = result;
    }
  </script>
